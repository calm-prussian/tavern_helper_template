    function populateCharacterData() {
      const all_variables = getAllVariables();

      /* --- 世界状态 --- */
      const time = _.get(all_variables, 'stat_data.世界.当前时间', '未知');
      const date = _.get(all_variables, 'stat_data.世界.当前日期', '未知');
      $('#val-time').text(time);
      $('#val-date').text(date);

      const affairsObj = _.get(all_variables, 'stat_data.世界.近期事务', {});
      const affairsText = Object.entries(affairsObj).map(([k, v]) => `${k}: ${v}`).join('; ') || '无';
      $('#val-affairs').text(affairsText);

      /* --- 格温内心与行为 --- */
      $('#val-thought').text(_.get(all_variables, 'stat_data.格温史黛西.内心反应', '暂无数据'));
      $('#val-action').text(_.get(all_variables, 'stat_data.格温史黛西.当前行为', '暂无数据'));

      /* --- 核心数值获取 --- */
      const affection = _.get(all_variables, 'stat_data.格温史黛西.好感度', 0);
      const trust = _.get(all_variables, 'stat_data.格温史黛西.信赖值', 0);
      const corruption = _.get(all_variables, 'stat_data.格温史黛西.堕落值', -100);
      const exposure = _.get(all_variables, 'stat_data.格温史黛西.露出倾向', 0);
      const awareness = _.get(all_variables, 'stat_data.格温史黛西.身份觉察度', 0);

      // 显示黄色数字
      $('#val-affection').text(affection);
      $('#val-trust').text(trust);
      $('#val-corruption').text(corruption);
      $('#val-exposure').text(exposure);
      $('#val-awareness').text(awareness);

      /* --- 计算电量高度 (按10%卡一格) --- */
      // 辅助函数：将数值按10为步长向下取整，转为0-100%
      const getBatPct = (val) => Math.min(100, Math.max(0, Math.floor(val / 10) * 10));

      // 常规0~100数值
      $('#fill-affection').css('height', getBatPct(affection) + '%');
      $('#fill-trust').css('height', getBatPct(trust) + '%');
      $('#fill-awareness').css('height', getBatPct(awareness) + '%');

      // 露出倾向 0~5：乘以20映射到0~100
      $('#fill-exposure').css('height', getBatPct(exposure * 20) + '%');

      // 堕落值 -100~100：跨度200，按比例映射到0~100 (比如 -100是0格，0是5格，100是满格)
      const corrPct = (corruption + 100) / 2;
      $('#fill-corruption').css('height', getBatPct(corrPct) + '%');

      /* --- 着装与身体状态表格 --- */
      const clothing = _.get(all_variables, 'stat_data.格温史黛西.着装', {});
      $('#val-cloth-top').text(_.get(clothing, '上装', '无'));
      $('#val-cloth-bottom').text(_.get(clothing, '下装', '无'));
      $('#val-cloth-inner').text(_.get(clothing, '内衣', '无'));
      $('#val-cloth-socks').text(_.get(clothing, '袜子', '无'));
      $('#val-cloth-shoes').text(_.get(clothing, '鞋子', '无'));
      $('#val-cloth-acc').text(_.get(clothing, '饰品', '无'));

      const organs = _.get(all_variables, 'stat_data.格温史黛西.性器官状态', {});
      $('#val-organ-pussy').text(_.get(organs, '小穴状态', '未知'));
      $('#val-organ-boobs').text(_.get(organs, '胸部状态', '未知'));
      $('#val-organ-ass').text(_.get(organs, '臀部状态', '未知'));
      $('#val-organ-feet').text(_.get(organs, '足部状态', '未知'));
    }

        $icon.text($icon.text() === '▼' ? '▶' : '▼');
      });
    }
