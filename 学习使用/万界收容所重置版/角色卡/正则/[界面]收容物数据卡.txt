/* ========== 工具 ========== */
function esc(s){if(!s)return '';var d=document.createElement('div');d.appendChild(document.createTextNode(s));return d.innerHTML;}
function rc(r){
  if(!r)return 'r-lv1';
  if(r.indexOf('LV5')>-1||r.indexOf('红')>-1)return 'r-lv5';
  if(r.indexOf('LV4')>-1||r.indexOf('金')>-1)return 'r-lv4';
  if(r.indexOf('LV3')>-1||r.indexOf('紫')>-1)return 'r-lv3';
  if(r.indexOf('LV2')>-1||r.indexOf('蓝')>-1)return 'r-lv2';
  return 'r-lv1';
}
function re(r){
  if(!r)return '⚪';
  if(r.indexOf('LV5')>-1||r.indexOf('红')>-1)return '🔴';
  if(r.indexOf('LV4')>-1||r.indexOf('金')>-1)return '🟡';
  if(r.indexOf('LV3')>-1||r.indexOf('紫')>-1)return '🟣';
  if(r.indexOf('LV2')>-1||r.indexOf('蓝')>-1)return '🔵';
  return '⚪';
}
function pe(p){
  if(!p)return '❓';
  if(p.indexOf('媚姬')>-1)return '👸';
  if(p.indexOf('肉壶')>-1)return '🔮';
  if(p.indexOf('雌兽')>-1)return '🐾';
  return '❓';
}
function routeT(r){
  if(r==='偏向淫荡')return '🔥 偏向淫荡';
  if(r==='偏向爱慕')return '💗 偏向爱慕';
  return '⚖️ 未定';
}
function pregE(s){
  if(!s||s==='正常')return '✅ 正常';
  if(s.indexOf('妊娠')>-1)return '🤰 '+s;
  if(s.indexOf('分娩')>-1)return '⚠️ '+s;
  return s;
}
/* ========== 构建器 ========== */
function mkRow(l,v,cls){return '<div class="sc-row"><span class="l">'+esc(l)+'</span><span class="v'+(cls?' '+cls:'')+'">'+esc(v||'—')+'</span></div>';}
function mkBar(l,em,val,mx,cls){
  var v=parseInt(val)||0;var m=parseInt(mx)||100;var pct=Math.min(100,Math.round(v/m*100));
  return '<div class="sc-bar '+cls+'"><div class="sc-bar-top"><span class="l">'+em+' '+esc(l)+'</span><span class="v">'+v+'/'+m+'</span></div><div class="sc-bar-track"><div class="sc-bar-fill" style="width:'+pct+'%"></div></div></div>';
}
function mkSec(icon,title,body,open){
  var c=open?'sc-sec open':'sc-sec';
  return '<div class="'+c+'"><div class="sc-sec-hdr"><span class="sc-sec-title">'+icon+' '+esc(title)+'</span><span class="sc-sec-arrow">▶</span></div><div class="sc-sec-body">'+body+'</div></div>';
}
function mkTagItem(raw,isCombo){
  var sep=raw.indexOf(':');
  var nm=sep>-1?raw.substring(0,sep).trim():raw;
  var ds=sep>-1?raw.substring(sep+1).trim():'';
  var icon=isCombo?'⚡':'🏷️';
  var ncls=isCombo?'sc-tag-item-name combo':'sc-tag-item-name';
  if(!ds)return '<div class="sc-tag-item"><div class="sc-tag-item-hdr"><span class="'+ncls+'">'+icon+' '+esc(nm)+'</span></div></div>';
  return '<div class="sc-tag-item"><div class="sc-tag-item-hdr"><span class="'+ncls+'">'+icon+' '+esc(nm)+'</span><span class="sc-tag-item-arr">▶</span></div><div class="sc-tag-item-desc">'+esc(ds)+'</div></div>';
}
function mkRtBlock(icon,label,text){
  return '<div class="sc-rt-block"><div class="sc-rt-label">'+icon+' '+esc(label)+'</div><div class="sc-rt-text">'+esc(text||'暂无数据')+'</div></div>';
}
/* ========== 解析 ========== */
function getMsg(){
  var c=getChatMessages(getCurrentMessageId());
  if(!c||c.length===0)return null;
  return c[0].message;
}
function pBlock(t,tag){var r=new RegExp('<'+tag+'>([\\s\\S]*?)</'+tag+'>');var m=t.match(r);return m?m[1]:'';}
function pKV(b){
  var res={};var r=/\[([^\|\]]+)\|([^\]]*)\]/g;var m;
  while((m=r.exec(b))!==null){res[m[1].trim()]=m[2].trim();}
  return res;
}
function pMulti(b,key){
  var items=[];var r=new RegExp('\\['+key+'\\|([^\\]]+)\\]','g');var m;
  while((m=r.exec(b))!==null){items.push(m[1].trim());}
  return items;
}
function pTasks(b){
  var tasks=[];var r=/\[任务\|([^\]]+)\]/g;var m;
  while((m=r.exec(b))!==null){
    var p=m[1].split('|');
    tasks.push({name:p[0]?p[0].trim():'',status:p[1]?p[1].trim():'',desc:p[2]?p[2].trim():''});
  }
  return tasks;
}
/* ========== 渲染 ========== */
function render(text){
  var cm=text.match(/<specimen_card>([\s\S]*?)<\/specimen_card>/);

  var card=cm[1];
  var basic=pKV(pBlock(card,'basic'));
  var appear=pKV(pBlock(card,'appearance'));

  var stats=pKV(pBlock(card,'stats'));
  var evo=pKV(pBlock(card,'evolution'));
  var tagRaw=pMulti(pBlock(card,'tags'),'标签');
  var comboRaw=pMulti(pBlock(card,'combos'),'效应');
  var hbkB=pBlock(card,'handbook');
  var hbkKV=pKV(hbkB);
  var tasks=pTasks(hbkB);

  /* 顶栏 */
  $('#hdr-name').text(basic['名称']||'未知收容物');
  if(basic['别称']){$('#hdr-alias').text('【'+basic['别称']+'】');}else{$('#hdr-alias').text('');}
  var badges='<span class="sc-hdr-badge '+rc(basic['稀有度'])+'">'+re(basic['稀有度'])+' '+(basic['稀有度']||'LV1')+'</span>';
  badges+='<span class="sc-hdr-badge">'+pe(basic['形态位阶'])+' '+(basic['形态位阶']||'未知')+'</span>';
  $('#hdr-badges').html(badges);
  /* ═══ 面板1：档案 ═══ */
  var p1='';

  var biH='<div class="sc-g2">'+
    mkRow('种族',basic['种族'])+

    mkRow('总点数',evo['进化总点数']||'0','ac')+

    }).join('');

/* ========== 事件 ========== */
function bindEvents(){
  $(document).on('click','.sc-sec-hdr',function(){$(this).closest('.sc-sec').toggleClass('open');});
  $(document).on('click','.sc-tag-item-hdr',function(e){e.stopPropagation();$(this).closest('.sc-tag-item').toggleClass('open');});
  $('.sc-nav-btn').on('click',function(){
    var p=$(this).data('p');
    $('.sc-nav-btn').removeClass('on');
    $(this).addClass('on');
    $('.sc-panel').removeClass('on');
    $('#panel-'+p).addClass('on');
  });
}
/* ========== 主函数 ========== */
function init(){
  try{
    var text=getMsg();
    if(!text){$('#panel-profile').html('<div class="sc-empty">❌ 无法获取消息内容</div>');return;}
    var ok=render(text);
    if(!ok){$('#panel-profile').html('<div class="sc-empty">❌ 未找到数据卡内容</div>');return;}
    bindEvents();
  }catch(err){
    console.error('收容物数据卡错误:',err);
    $('#panel-profile').html('<div class="sc-empty">❌ 加载失败：'+esc(err.message)+'</div>');
  }
}
$(function(){init();});
