function rc(r){
  if(!r)return'r1';
  var s=String(r);
  if(s.includes('红')||s.includes('LV5')||s==='5')return'r5';
  if(s.includes('金')||s.includes('LV4')||s==='4')return'r4';
  if(s.includes('紫')||s.includes('LV3')||s==='3')return'r3';
  if(s.includes('蓝')||s.includes('LV2')||s==='2')return'r2';
  return'r1';
}
function re(r){
  if(!r)return'⚪';
  var s=String(r);

}

}
