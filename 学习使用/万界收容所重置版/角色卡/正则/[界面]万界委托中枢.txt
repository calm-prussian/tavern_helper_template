        function getMessageData() {
            var chatMessages = getChatMessages(getCurrentMessageId());

                console.error("无法获取消息内容");
                return null;

        /* ========== 解析数据 ========== */
        function parseHub(text) {
            var data = {
                level: '',
                levelNum: 1,
                sp: '',
                habitat: '',
                commissions: [],
                market: [],
                captureInfo: ''
            };
            var hubM = text.match(/<commission_hub>([\s\S]*?)<\/commission_hub>/);
            if (!hubM || !hubM[1]) return data;
            var c = hubM[1];
            var lm = c.match(/\[收容所等级\|([^\]]*)\]/);
            if (lm) data.level = lm[1].trim();
            var sm = c.match(/\[当前源点\|([^\]]*)\]/);
            if (sm) data.sp = sm[1].trim();
            var hm = c.match(/\[栖息地\|([^\]]*)\]/);
            if (hm) data.habitat = hm[1].trim();
            var ln = data.level.match(/LV(\d)/);
            data.levelNum = ln ? parseInt(ln[1]) : 1;
            /* 公共委托 */
            var pm = c.match(/<public>([\s\S]*?)<\/public>/);
            if (pm && pm[1]) {
                var pLines = pm[1].trim().split('\n');
                for (var i = 0; i < pLines.length; i++) {
                    var rm = pLines[i].match(/\[委托\|([^\]]*)\]/);
                    if (rm) {
                        var p = rm[1].split('|');
                        data.commissions.push({
                            id: (p[0] || '').trim(),
                            client: (p[1] || '').trim(),
                            clientBg: (p[2] || '').trim(),
                            race: (p[3] || '').trim(),
                            desc: (p[4] || '').trim(),
                            rarity: (p[5] || '').trim(),
                            req: (p[6] || '').trim(),
                            reward: (p[7] || '').trim()
                        });
                    }
                }
            }
            /* 黑市 */
            var mm = c.match(/<market>([\s\S]*?)<\/market>/);
            if (mm && mm[1]) {
                var mLines = mm[1].trim().split('\n');
                for (var j = 0; j < mLines.length; j++) {
                    var sm2 = mLines[j].match(/\[商品\|([^\]]*)\]/);
                    if (sm2) {
                        var q = sm2[1].split('|');
                        data.market.push({
                            id: (q[0] || '').trim(),
                            name: (q[1] || '').trim(),
                            race: (q[2] || '').trim(),
                            world: (q[3] || '').trim(),
                            desc: (q[4] || '').trim(),
                            rarity: (q[5] || '').trim(),
                            combat: (q[6] || '').trim(),
                            price: (q[7] || '').trim()
                        });
                    }
                }
            }
            /* 捕捉 */
            var cm = c.match(/<capture>([\s\S]*?)<\/capture>/);
            if (cm && cm[1]) {
                var cl = cm[1].match(/\[捕捉\|([^\]]*)\]/);
                if (cl) data.captureInfo = cl[1].trim();
            }
            return data;
        }

        function badgeCls(r) {
            if (r.indexOf('5') !== -1 || r.indexOf('红') !== -1) return 'badge-lv5';
            if (r.indexOf('4') !== -1 || r.indexOf('金') !== -1) return 'badge-lv4';
            if (r.indexOf('3') !== -1 || r.indexOf('紫') !== -1) return 'badge-lv3';
            if (r.indexOf('2') !== -1 || r.indexOf('蓝') !== -1) return 'badge-lv2';
            return 'badge-lv1';
        }

        /* ========== Tab切换 ========== */
        function swTab(id, btn) {
            if (!btn || btn.classList.contains('locked')) return;
            var bodies = document.querySelectorAll('.tab-body');
            for (var i = 0; i < bodies.length; i++) { bodies[i].classList.remove('active'); }
            var el = document.getElementById('tb-' + id);
            if (el) el.classList.add('active');
            var btns = document.querySelectorAll('.tab-btn');
            for (var j = 0; j < btns.length; j++) { btns[j].classList.remove('active'); }
            btn.classList.add('active');
        }

            }
