        /* ========== 获取消息内容 ========== */
        function getMessageData() {
            var chatMessages = getChatMessages(getCurrentMessageId());
            if (!chatMessages || chatMessages.length === 0) {
                console.error("无法获取消息内容");
                return null;
            }
            return chatMessages[0].message;
        }
        /* ========== 解析数据 ========== */
        function parsePanel(text) {
            var data = {
                date: '',
                sp: '',
                habitat: '',
                offspring: []
            };
            var wm = text.match(/<offspring_panel>([\s\S]*?)<\/offspring_panel>/);
            if (!wm || !wm[1]) return data;
            var c = wm[1];
            var dm = c.match(/\[游戏日期\|([^\]]*)\]/);
            if (dm) data.date = dm[1].trim();
            var sm = c.match(/\[当前源点\|([^\]]*)\]/);
            if (sm) data.sp = sm[1].trim();
            var hm = c.match(/\[栖息地\|([^\]]*)\]/);
            if (hm) data.habitat = hm[1].trim();
            /* 解析子代列表 */
            var om = c.match(/<offspring>([\s\S]*?)<\/offspring>/);
            if (om && om[1]) {
                var lines = om[1].trim().split('\n');
                for (var i = 0; i < lines.length; i++) {
                    var lm = lines[i].match(/\[子代\|([^\]]*)\]/);
                    if (lm) {
                        var p = lm[1].split('|');
                        var rawTags = (p[5] || '').trim();
                        var tagArr = rawTags ? rawTags.split(',').map(function(t) { return t.trim(); }).filter(function(t) { return t; }) : [];
                        data.offspring.push({
                            id:       (p[0] || '').trim(),
                            mother:   (p[1] || '').trim(),
                            rarity:   (p[2] || '').trim(),
                            race:     (p[3] || '').trim(),
                            tags:     tagArr,
                            costCult: (p[6] || '').trim(),
                            costSell: (p[7] || '').trim()
                        });
                    }
                }
            }
            return data;
        }
        /* ========== 稀有度样式 ========== */
        function rarityClass(r) {
            if (r.indexOf('5') !== -1 || r.indexOf('红') !== -1) return 'r-lv5';
            if (r.indexOf('4') !== -1 || r.indexOf('金') !== -1) return 'r-lv4';
            if (r.indexOf('3') !== -1 || r.indexOf('紫') !== -1) return 'r-lv3';
            if (r.indexOf('2') !== -1 || r.indexOf('蓝') !== -1) return 'r-lv2';
            return 'r-lv1';
        }
        /* ========== 转义 ========== */
        function esc(s) {
            return (s || '').replace(/\\/g, '\\\\').replace(/'/g, "\\'");
        }
        /* ========== 渲染 ========== */
        function renderPanel(d) {
            var h = '';
            h += '<div class="panel-title">🧬 培养房 · 子代处置 🧬</div>';
            h += '<div class="panel-sub">OFFSPRING DISPOSAL SYSTEM</div>';
            /* 状态栏 */
            var parts = (d.habitat || '0/0').split('/');
            var used = parseInt(parts[0]) || 0;
            var total = parseInt(parts[1]) || 0;
            var habitatWarn = (used >= total) ? ' warn' : '';
            h += '<div class="status-bar">';
            h += '<div class="st-item">📅 <span class="st-val">' + (d.date || '第—日') + '</span></div>';
            h += '<div class="st-item">💠 源点 <span class="st-val">' + (d.sp || '0') + '</span></div>';
            h += '<div class="st-item">🏠 栖息地 <span class="st-val' + habitatWarn + '">' + (d.habitat || '0/0') + '</span></div>';
            h += '<div class="st-item">🍼 待处置 <span class="st-val">' + d.offspring.length + '</span></div>';
            h += '</div>';
            /* 说明 */
            h += '<div class="panel-guide">';
            h += '💡 点击 <b>培育为收容物</b> 后，在对话框中告诉我你想继承哪些标签（最多3个）。<br>';
            h += '💰 点击 <b>作为资源出售</b> 将直接结算源点，子代条目随即移除。';
            h += '</div>';
            /* 子代列表 */
            h += '<div class="offspring-section">';
            h += '<div class="section-title">— 待处置子代列表 —</div>';
            if (d.offspring.length === 0) {
                h += '<div class="empty">';
                h += '<div class="empty-ico">🍼</div>';
                h += '<div class="empty-msg">培养房当前无待处置子代</div>';
                h += '<div class="empty-sub">收容物分娩后将自动出现在此列表</div>';
                h += '</div>';
            } else {
                for (var i = 0; i < d.offspring.length; i++) {
                    var o = d.offspring[i];
                    var rc = rarityClass(o.rarity || '');
                    h += '<div class="card">';
                    /* 卡头 */
                    h += '<div class="card-head">';
                    h += '<div class="card-title-wrap">';
                    h += '<span class="card-id">#' + (o.id || i + 1) + '</span>';
                    h += '<span class="card-name">🐣 ' + (o.mother || '未知') + ' 之子代</span>';
                    h += '</div>';
                    h += '<span class="badge ' + rc + '">' + (o.rarity || 'LV1白色') + '</span>';
                    h += '</div>';
                    /* 信息网格 */
                    h += '<div class="info-grid">';
                    h += '<div class="field"><span class="field-k">母体：</span><span class="field-v">' + (o.mother || '—') + '</span></div>';
                    h += '<div class="field"><span class="field-k">种族：</span><span class="field-v">' + (o.race || '—') + '</span></div>';
                    h += '</div>';
                    /* 可继承标签 */
                    h += '<div class="tags-label">🏷️ 可继承标签（最多选3个）：</div>';
                    h += '<div class="tags-wrap">';
                    if (o.tags && o.tags.length > 0) {
                        for (var j = 0; j < o.tags.length; j++) {
                            h += '<span class="tag-chip">' + o.tags[j] + '</span>';
                        }
                    } else {
                        h += '<span class="tag-chip" style="color:#1a3a28;border-color:rgba(0,255,136,0.05);">暂无可继承标签</span>';
                    }
                    h += '</div>';
                    /* 底部操作 */
                    h += '<div class="card-foot">';
                    h += '<div class="cost-info">';
                    h += '<div class="cost-row"><span class="cost-k">培育费用：</span><span class="cost-cultivate">💠 ' + (o.costCult || '—') + '</span></div>';
                    h += '<div class="cost-row"><span class="cost-k">出售收益：</span><span class="cost-sell">💰 ' + (o.costSell || '—') + '</span></div>';
                    h += '</div>';
                    h += '<div class="btn-group">';
                    h += '<button class="btn btn-green" onclick="doCultivate(\'' + esc(o.id) + '\',\'' + esc(o.mother) + '\',\'' + esc(o.race) + '\')">🌱 培育为收容物</button>';
                    h += '<button class="btn btn-yellow" onclick="doSell(\'' + esc(o.id) + '\',\'' + esc(o.mother) + '\',\'' + esc(o.costSell) + '\')">💰 作为资源出售</button>';
                    h += '</div>';
                    h += '</div>';
                    h += '</div>';
                }
            }
            h += '</div>';
            document.getElementById('panel').innerHTML = h;
        }
        /* ========== 培育子代 ========== */
        function doCultivate(id, mother, race) {
            if (typeof triggerSlash === 'function') {
                triggerSlash('/send 培育子代：编号' + id + '，母体「' + mother + '」，种族「' + race + '」。请列出可继承标签，我来选择最多3个。|/trigger');
            }
        }
        /* ========== 出售子代 ========== */
        function doSell(id, mother, price) {
            if (typeof triggerSlash === 'function') {
                triggerSlash('/send 出售子代：编号' + id + '，母体「' + mother + '」，预计收益' + price + '，确认出售。|/trigger');
            }
        }
        /* ========== 主函数 ========== */
        function init() {
            try {
                var msg = getMessageData();
                if (!msg) {
                    document.getElementById('panel').innerHTML = '<div class="loading">❌ 无法连接培养房</div>';
                    return;
                }
                var data = parsePanel(msg);
                renderPanel(data);
            } catch (err) {
                console.error("渲染错误:", err);
                document.getElementById('panel').innerHTML = '<div class="loading">❌ 系统故障：' + err.message + '</div>';
            }
        }
        $(function() { init(); });
    </script>
