<script>
/* ========== 全局状态 ========== */
var selected = {};
var allTags = [];
var targetName = '';
var hasSent = false;
/* ========== 工具 ========== */
function esc(s){if(!s)return '';var d=document.createElement('div');d.appendChild(document.createTextNode(s));return d.innerHTML;}
var catMeta = {
  '肉体塑形': {icon:'🍑', label:'肉体塑形'},
  '感官开发': {icon:'✨', label:'感官开发'},
  '性格塑造': {icon:'🧠', label:'性格塑造'},
  '能力觉醒': {icon:'⚡', label:'能力觉醒'},
  '常识修改': {icon:'🔄', label:'常识修改'},
  '能力强化': {icon:'⬆️', label:'能力强化'},
  '能力弱化': {icon:'⬇️', label:'能力弱化'}
};
var catOrder = ['肉体塑形','感官开发','性格塑造','能力觉醒','常识修改','能力强化','能力弱化'];
/* ========== 获取消息 ========== */
function getMsg(){
  var c = getChatMessages(getCurrentMessageId());
  if(!c || c.length === 0) return null;
  return c[0].message;
}
/* ========== 解析 ========== */
function parsePool(text){
  var m = text.match(/<tag_pool>([\s\S]*?)<\/tag_pool>/);
  if(!m) return null;
  var content = m[1];
  var targetMatch = content.match(/\[收容物\|([^\]]+)\]/);
  if(targetMatch) targetName = targetMatch[1].trim();
  var tags = [];
  var re = /\[(肉体塑形|感官开发|性格塑造|能力觉醒|常识修改|能力强化|能力弱化)\|([^\|]+)\|([^\|]+)\|([^\]]+)\]/g;
  var tm;
  while((tm = re.exec(content)) !== null){
    tags.push({
      cat: tm[1].trim(),
      rarity: tm[2].trim(),
      name: tm[3].trim(),
      desc: tm[4].trim()
    });
  }
  return tags;
}
/* ========== 渲染 ========== */
function render(tags){
  allTags = tags;
  var grouped = {};
  catOrder.forEach(function(c){ grouped[c] = []; });
  tags.forEach(function(t){
    if(grouped[t.cat]) grouped[t.cat].push(t);
  });
  $('#tp-target').html('🎯 ' + esc(targetName || '未知收容物'));
  var html = '';
  catOrder.forEach(function(cat){
    var items = grouped[cat];
    if(items.length === 0) return;
    var meta = catMeta[cat] || {icon:'📌', label: cat};
    html += '<div class="tp-group">';
    html += '<div class="tp-group-hdr">';
    html += '<span class="cat-icon">' + meta.icon + '</span>';
    html += '<span>' + esc(meta.label) + '</span>';
    html += '<span class="tp-group-badge">' + items.length + '</span>';
    html += '</div>';
    items.forEach(function(t, i){
      var uid = cat + '_' + i;
      var rCls = 'r-' + t.rarity;
      html += '<div class="tp-card" data-uid="' + esc(uid) + '" data-name="' + esc(t.name) + '">';
      html += '<div class="tp-card-hdr">';
      html += '<div class="tp-chk">✓</div>';
      html += '<span class="tp-card-name">' + esc(t.name) + '</span>';
      html += '<span class="tp-card-rarity ' + rCls + '">' + esc(t.rarity) + '</span>';
      html += '<span class="tp-card-arrow">▶</span>';
      html += '</div>';
      html += '<div class="tp-card-desc">' + esc(t.desc) + '</div>';
      html += '</div>';
    });
    html += '</div>';
  });
  $('#tp-body').html(html);
  updateCount();
}
/* ========== 交互 ========== */
function toggleSelect(uid, name){
  if(hasSent) return;
  if(selected[uid]){
    delete selected[uid];
  } else {
    selected[uid] = name;
  }
  updateCardState(uid);
  updateCount();
}
function updateCardState(uid){
  var $card = $('.tp-card[data-uid="' + uid + '"]');
  if(selected[uid]){
    $card.addClass('selected');
  } else {
    $card.removeClass('selected');
  }
}
function toggleExpand(uid){
  var $card = $('.tp-card[data-uid="' + uid + '"]');
  $card.toggleClass('expanded');
}
function updateCount(){
  var count = Object.keys(selected).length;
  $('#tp-count').text(count);
  if(count > 0){
    $('#btn-confirm').removeClass('disabled');
  } else {
    $('#btn-confirm').addClass('disabled');
  }
}
function clearAll(){
  if(hasSent) return;
  selected = {};
  $('.tp-card').removeClass('selected');
  updateCount();
}
function selectAll(){
  if(hasSent) return;
  selected = {};
  $('.tp-card').each(function(){
    var uid = $(this).data('uid');
    var name = $(this).data('name');
    selected[uid] = name;
    $(this).addClass('selected');
  });
  updateCount();
}
function confirmSend(){
  if(hasSent) return;
  var names = [];
  Object.keys(selected).forEach(function(uid){
    names.push(selected[uid]);
  });
  if(names.length === 0) return;
  hasSent = true;
  var msg = '为【' + targetName + '】选择以下标签进行培育：' + names.join('、');
  /* 显示已发送状态 */
  $('#tp-action-bar').html(
    '<div class="tp-sent">' +
      '<div class="tp-sent-icon">✅</div>' +
      '<div class="tp-sent-text">已提交 ' + names.length + ' 个标签</div>' +
      '<div class="tp-sent-detail">' + names.map(function(n){return '【' + esc(n) + '】';}).join(' ') + '</div>' +
    '</div>'
  );
  /* 锁定所有卡片 */
  $('.tp-card').css('opacity', '0.5').css('pointer-events', 'none');
  $('.tp-card.selected').css('opacity', '1');
  if(typeof triggerSlash === 'function'){
    triggerSlash('/send ' + msg + '|/trigger');
  }
}
/* ========== 事件绑定 ========== */
function bindEvents(){
  /* 点击卡片头部 */
  $(document).on('click', '.tp-card-hdr', function(e){
    var $card = $(this).closest('.tp-card');
    var uid = $card.data('uid');
    var name = $card.data('name');
    /* 判断点击区域：复选框区域 vs 展开区域 */
    var $chk = $(this).find('.tp-chk');
    var $arrow = $(this).find('.tp-card-arrow');
    var clickX = e.pageX;
    var arrowLeft = $arrow.offset().left;
    /* 如果点击位置在箭头附近（右侧30px范围），则展开/折叠 */
    if(clickX >= arrowLeft - 15){
      toggleExpand(uid);
    } else {
      /* 否则切换选中 */
      toggleSelect(uid, name);
    }
  });
}
/* ========== 主函数 ========== */
function init(){
  try{
    var text = getMsg();
    if(!text){
      $('#tp-body').html('<div class="tp-empty">❌ 无法获取消息内容</div>');
      return;
    }
    var tags = parsePool(text);
    if(!tags || tags.length === 0){
      $('#tp-body').html('<div class="tp-empty">❌ 未找到标签数据</div>');
      return;
    }
    render(tags);
    bindEvents();
  }catch(err){
    console.error('标签生成池错误:', err);
    $('#tp-body').html('<div class="tp-empty">❌ 加载失败：' + esc(err.message) + '</div>');
  }
}
$(function(){ init(); });
</script>
</body>
</html>
