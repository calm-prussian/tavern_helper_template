function pickGender(el) {
  var btns = document.querySelectorAll(".cc-g-btn");
  for (var i = 0; i < btns.length; i++) {
    btns[i].classList.remove("on");
  }
  el.classList.add("on");
  chosenGender = el.getAttribute("data-g") || "";
  checkReady();

function pickLv(el) {
  var tabs = document.querySelectorAll(".cc-lv-tab");
  for (var i = 0; i < tabs.length; i++) {

  el.classList.add("on");
  chosenLv = el.getAttribute("data-lv") || "";
  renderDetail(chosenLv);
  checkReady();
}

  if (d.pool.r > 0) html += '<span class="cc-pl-item"><span class="cc-pl-dot" style="background:#ef4444"></span>红 ' + d.pool.r + '%</span>';
  html += '</div>';
  html += '</div>';
  html += '</div>'; /* 关闭 cc-det-grid */
  document.getElementById("lvDetail").innerHTML = html;
}
/* ===== 就绪检查 ===== */
function checkReady() {
  var nameVal = (document.getElementById("inpName").value || "").trim();
  var btn = document.getElementById("confirmBtn");
  var hint = document.getElementById("hintText");
  if (nameVal.length > 0 && chosenGender !== "" && chosenLv !== "") {
    btn.classList.remove("off");
    btn.classList.add("ready");
    hint.textContent = "✅ 信息已就绪，点击启动认证";
    hint.style.color = "#2a5a6a";
  } else {
    btn.classList.add("off");
    btn.classList.remove("ready");
    var missing = [];
    if (nameVal.length === 0) missing.push("称呼");
    if (chosenGender === "") missing.push("性别");
    if (chosenLv === "") missing.push("阶位");
    hint.textContent = "请完成：" + missing.join("、");
    hint.style.color = "#1e2e42";
  }
}

function submitCreate() {
  var nameVal = (document.getElementById("inpName").value || "").trim();
  if (nameVal.length === 0 || chosenGender === "" || chosenLv === "") return;
  var lvInfo = LV_DATA[chosenLv] || {};
  var lvName = lvInfo.name || ("LV" + chosenLv);
  var msg = "角色创建：我的名字是" + nameVal + "，性别" + chosenGender + "，选择初始阶位LV" + chosenLv + "（" + lvName + "）。初始源点" + (lvInfo.sp || "0") + " SP，栖息地容量" + (lvInfo.slots || "0") + "单元。请以此身份开始游戏。";
  if (typeof triggerSlash === "function") {
    triggerSlash("/send " + msg + "|/trigger");
  }
}
