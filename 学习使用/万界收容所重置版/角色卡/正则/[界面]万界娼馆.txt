/* ========== 获取消息 ========== */
function getMsg(){
    var msgs=getChatMessages(getCurrentMessageId());
    if(!msgs||msgs.length===0)return null;
    return msgs[0].message;

/* ========== 提取标签内容 ========== */
function extractTag(text,tag){
    var re=new RegExp('<'+tag+'>([\\s\\S]*?)</'+tag+'>');
    var m=text.match(re);
    return m?m[1]:'';
}
/* ========== 解析 [key|val] 行 ========== */
function parseBrackets(text){
    var result=[];
    var lines=text.split('\n');
    for(var i=0;i<lines.length;i++){
        var line=lines[i].trim();
        if(line.charAt(0)==='['&&line.charAt(line.length-1)===']'){
            var inner=line.substring(1,line.length-1);
            var parts=inner.split('|');
            result.push(parts);
        }
    }
    return result;
}
/* ========== 查找字段值 ========== */
function findField(rows,key){
    for(var i=0;i<rows.length;i++){
        if(rows[i][0]&&rows[i][0].trim()===key){
            if(rows[i].length===2)return rows[i][1].trim();
            var rest=[];
            for(var j=1;j<rows[i].length;j++){rest.push(rows[i][j].trim());}
            return rest.join('|');
        }
    }
    return '';
}
/* ========== 查找所有同名行 ========== */
function findAll(rows,key){
    var arr=[];
    for(var i=0;i<rows.length;i++){
        if(rows[i][0]&&rows[i][0].trim()===key){
            arr.push(rows[i]);
        }
    }
    return arr;
}

function renderMonitor(text){
    var inner=extractTag(text,'monitor');
    var rows=parseBrackets(inner);
    var slotId=findField(rows,'展位')||'--';
    var specName=findField(rows,'收容物')||'--';
    var guest=findField(rows,'来客')||'--';
    var sceneRaw=extractTag(inner,'scene')||'';
    /* 处理正文 */
    var paragraphs=sceneRaw.split(/\n\s*\n/);
    var sceneHtml='';
    for(var i=0;i<paragraphs.length;i++){
        var p=paragraphs[i].trim();
        if(p){
            /* 单换行也处理 */
            var subLines=p.split(/\n/);
            for(var j=0;j<subLines.length;j++){
                var sl=subLines[j].trim();
                if(sl){
                    sceneHtml+='<p>'+sl+'</p>';
                }
            }
        }
    }
    if(!sceneHtml)sceneHtml='<p style="color:#9b8abf;text-align:center;">监视信号接收中...</p>';

function init(){
    var root=document.getElementById('exRoot');
    try{
        var msg=getMsg();
        if(!msg){
            root.innerHTML='<div class="ex-error">❌ 无法获取消息内容</div>';
            return;
        }
        /* 提取 exhibition 标签 */
        var exContent=extractTag(msg,'exhibition');
        if(!exContent){
            root.innerHTML='<div class="ex-error">❌ 未找到展厅数据</div>';
            return;
        }
        var panel=detectPanel(exContent);
        var rendered='';
        if(panel==='hall'){
            rendered=renderHall(exContent);
        }else if(panel==='visitor'){
            rendered=renderVisitor(exContent);
        }else if(panel==='settlement'){
            rendered=renderSettlement(exContent);
        }else if(panel==='monitor'){
            rendered=renderMonitor(exContent);
        }else{
            root.innerHTML='<div class="ex-error">❌ 未识别的面板类型</div>';
            return;
        }
        root.innerHTML=rendered;
    }catch(e){
        console.error("万界展厅渲染错误:",e);
        root.innerHTML='<div class="ex-error">❌ 渲染失败：'+e.message+'</div>';
    }
}
$(function(){init();});
