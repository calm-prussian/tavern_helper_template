        /* ========== 获取消息内容 ========== */
        function getMessageData() {
            var chatMessages = getChatMessages(getCurrentMessageId());
            if (!chatMessages || chatMessages.length === 0) {
                console.error("无法获取消息内容");
                return null;
            }
            return chatMessages[0].message;
        }
        /* ========== 解析数据 ========== */
        function parseForge(text) {
            var data = {
                target: '',
                tagCount: '',
                triggerNum: '',
                comboCount: '',
                tags: [],
                tip: ''
            };
            var wm = text.match(/<combo_forge>([\s\S]*?)<\/combo_forge>/);
            if (!wm || !wm[1]) return data;
            var c = wm[1];
            var tm = c.match(/\[目标\|([^\]]*)\]/);
            if (tm) data.target = tm[1].trim();
            var nm = c.match(/\[普通标签数\|([^\]]*)\]/);
            if (nm) data.tagCount = nm[1].trim();
            var fnm = c.match(/\[触发倍数\|([^\]]*)\]/);
            if (fnm) data.triggerNum = fnm[1].trim();
            var cm = c.match(/\[已有组合效应数\|([^\]]*)\]/);
            if (cm) data.comboCount = cm[1].trim();
            var tipm = c.match(/\[提示\|([^\]]*)\]/);
            if (tipm) data.tip = tipm[1].trim();
            /* 提取 --- 之间的标签列表 */
            var blockM = c.match(/---\s*([\s\S]*?)\s*---/);
            if (blockM && blockM[1]) {
                var lines = blockM[1].trim().split('\n');
                for (var i = 0; i < lines.length; i++) {
                    var line = lines[i].trim();
                    if (!line) continue;
                    /* 格式：序号. 标签名 | 稀有度 | 描述 */
                    var lm = line.match(/^(\d+)\.\s*([^|]+)\|([^|]+)\|(.+)$/);
                    if (lm) {
                        data.tags.push({
                            num: lm[1].trim(),
                            name: lm[2].trim(),
                            rarity: lm[3].trim(),
                            desc: lm[4].trim()
                        });
                    }
                }
            }
            return data;
        }
        /* ========== 稀有度样式 ========== */
        function rarityClass(r) {
            if (r.indexOf('红') !== -1) return 'r-red';
            if (r.indexOf('金') !== -1) return 'r-gold';
            if (r.indexOf('紫') !== -1) return 'r-purple';
            if (r.indexOf('蓝') !== -1) return 'r-blue';
            return 'r-white';
        }
        /* ========== 复制标签名 ========== */
        function copyTag(btn, name) {
            if (navigator.clipboard) {
                navigator.clipboard.writeText(name).then(function() {
                    btn.textContent = '✓ 已复制';
                    btn.classList.add('copied');
                    setTimeout(function() {
                        btn.textContent = '复制';
                        btn.classList.remove('copied');
                    }, 1500);
                });
            } else {
                var ta = document.createElement('textarea');
                ta.value = name;
                document.body.appendChild(ta);
                ta.select();
                document.execCommand('copy');
                document.body.removeChild(ta);
                btn.textContent = '✓ 已复制';
                btn.classList.add('copied');
                setTimeout(function() {
                    btn.textContent = '复制';
                    btn.classList.remove('copied');
                }, 1500);
            }
        }
        /* ========== 渲染 ========== */
        function renderForge(d) {
            var h = '';
            h += '<div class="forge-title">⚗️ 组合效应合成炉 ⚗️</div>';
            h += '<div class="forge-subtitle">COMBO EFFECT FORGE</div>';
            /* 状态栏 */
            h += '<div class="status-bar">';
            h += '<div class="st-item">🧬 收容物 <span class="st-val">' + (d.target || '—') + '</span></div>';
            h += '<div class="st-item">🏷️ 普通标签 <span class="st-val highlight">' + (d.tagCount || '0') + '</span></div>';
            h += '<div class="st-item">✨ 组合效应 <span class="st-val">' + (d.comboCount || '0') + '</span></div>';
            h += '<div class="st-item">⚡ 触发节点 <span class="st-val">' + (d.triggerNum || '—') + '</span></div>';
            h += '</div>';
            /* 说明 */
            h += '<div class="forge-guide">';
            h += '💡 从下方标签中<b>选择3个</b>进行融合，直接在对话框中告诉我你想合成的3个标签名即可。<br>';
            h += '⚠️ 合成将消耗所选的3个普通标签，生成1个更强大的组合效应，进化点数维持不变。';
            h += '</div>';
            /* 标签列表 */
            h += '<div class="tag-section">';
            h += '<div class="tag-section-title">— 当前普通标签列表 —</div>';
            if (d.tags.length === 0) {
                h += '<div class="empty">暂无普通标签</div>';
            } else {
                h += '<div class="tag-list">';
                for (var i = 0; i < d.tags.length; i++) {
                    var t = d.tags[i];
                    var rc = rarityClass(t.rarity || '');
                    h += '<div class="tag-row">';
                    h += '<span class="tag-num">' + t.num + '.</span>';
                    h += '<div class="tag-body">';
                    h += '<div class="tag-head">';
                    h += '<span class="tag-name">' + (t.name || '') + '</span>';
                    h += '<span class="rarity-badge ' + rc + '">' + (t.rarity || '') + '</span>';
                    h += '</div>';
                    h += '<div class="tag-desc">' + (t.desc || '') + '</div>';
                    h += '</div>';
                    h += '<button class="tag-copy-btn" onclick="copyTag(this,\'' + (t.name || '').replace(/'/g, "\\'") + '\')">复制</button>';
                    h += '</div>';
                }
                h += '</div>';
            }
            h += '</div>';
            /* 提示 */
            if (d.tip) {
                h += '<div class="forge-tip">📝 <span>' + d.tip + '</span></div>';
            }
            document.getElementById('forge').innerHTML = h;
        }
        /* ========== 主函数 ========== */
        function init() {
            try {
                var msg = getMessageData();
                if (!msg) {
                    document.getElementById('forge').innerHTML = '<div class="loading">❌ 无法连接合成炉</div>';
                    return;
                }
                var data = parseForge(msg);
                renderForge(data);
            } catch (err) {
                console.error("渲染错误:", err);
                document.getElementById('forge').innerHTML = '<div class="loading">❌ 系统故障：' + err.message + '</div>';
            }
        }
        $(function() { init(); });
    </script>
