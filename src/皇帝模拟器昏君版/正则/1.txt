// 处理多个心理活动
const thoughtsText = '$9';
const thoughtsContainer = document.getElementById('thoughts-content');

// 使用正则匹配所有心理活动
const thoughtMatches = thoughtsText.match(/\[([^：]+)：([^\]]+)\]/g);

if (thoughtMatches) {
  thoughtMatches.forEach(match => {
    const [, name, thought] = match.match(/\[([^：]+)：([^\]]+)\]/);
    const thoughtDiv = document.createElement('div');
    thoughtDiv.className = 'character-thought';
    thoughtDiv.innerHTML = `<div class="thought-content"><strong>${name}：</strong>${thought}</div>`;
    thoughtsContainer.appendChild(thoughtDiv);
  });
} else {
  thoughtsContainer.innerHTML = '<div class="character-thought"><div class="thought-content">暂无心理活动</div></div>';
}
